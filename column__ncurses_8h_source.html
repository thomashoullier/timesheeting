129  std::string query_row_input (int input_row) {
130  std::string input_buffer {};
131  // Move cursor to beginning of line.
132  wmove(win, input_row, 0);
133  // Clear the display line.
134  wclrtoeol(win);
135  bool user_wants_to_input = true;
136  while (user_wants_to_input) { // Item input loop.
137  auto ch = wgetch(win);
138  switch (ch) {
139  case '\n': // User validates the input.
140  user_wants_to_input = false;
141  break;
142  case 27: // User wants to cancel.
143  input_buffer.clear();
144  user_wants_to_input = false;
145  break;
146  case 127: // Erase last character
147  if (!input_buffer.empty()) {
148  input_buffer.pop_back();
149  // Remove character from screen
150  int y, x;
151  getyx(win, y, x);
152  auto xmax = getmaxx(win);
153  if (x == 0) { // jump to the end of the previous line.
154  wmove(win, y-1, xmax-1);
155  } else {
156  wmove(win, y, x - 1);
157  }
158  wdelch(win);
159  }
160  break;
161  default: // Gets added to item.
162  input_buffer.push_back(ch);
163  waddch(win, ch);
164  break;
165  }
166  }
167  // Cleanup the input row.
168  wmove(win, input_row, 0);
169  wclrtoeol(win);
170  return input_buffer;
171  }
