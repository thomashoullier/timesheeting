157  std::string query_row_input () {
158  std::string input_buffer {};
159  // Clear the display line.
160  wclrtoeol(win);
161  bool user_wants_to_input = true;
162  while (user_wants_to_input) { // Item input loop.
163  auto ch = wgetch(win);
164  switch (ch) {
165  case '\n': // User validates the input.
166  user_wants_to_input = false;
167  break;
168  case 27: // User wants to cancel.
169  input_buffer.clear();
170  user_wants_to_input = false;
171  break;
172  case 127: // Erase last character
173  if (!input_buffer.empty()) {
174  input_buffer.pop_back();
175  // Remove character from screen
176  int y, x;
177  getyx(win, y, x);
178  auto xmax = getmaxx(win);
179  if (x == 0) { // jump to the end of the previous line.
180  wmove(win, y-1, xmax-1);
181  } else {
182  wmove(win, y, x - 1);
183  }
184  wdelch(win);
185  }
186  break;
187  default: // Gets added to item.
188  input_buffer.push_back(ch);
189  waddch(win, ch);
190  break;
191  }
192  }
193  return input_buffer;
